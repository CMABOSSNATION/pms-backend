// prisma/schema.prisma
// Database: SQLite — fully offline, no internet required
// Optimised for Render deployment (persistent disk) + offline-first PWA sync

generator client {
  provider      = "prisma-client-js"
  // binaryTargets ensures the Prisma engine works on Render (linux-musl)
  // AND on local dev machines (Windows / macOS / Linux x64)
  binaryTargets = ["native", "linux-musl-openssl-3.0.x", "debian-openssl-3.0.x"]
}

datasource db {
  provider = "sqlite"
  // On Render: set DATABASE_URL = "file:/data/prison.db"  (persistent disk mounted at /data)
  // Locally:   set DATABASE_URL = "file:./dev.db"
  url      = env("DATABASE_URL")
}

// ─── USERS (Staff login accounts) ─────────────────────────────────────────────
model User {
  id          String   @id @default(cuid())
  name        String
  email       String   @unique
  password    String
  role        String   @default("GUARD") // WARDEN | GUARD | MEDICAL | ADMIN | LEGAL | INVENTORY
  badgeNumber String   @unique
  shift       String   @default("DAY")
  isActive    Boolean  @default(true)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  incidents          Incident[]          @relation("ReportedBy")
  scans              ScanLog[]           @relation("ScannedBy")
  auditLogs          AuditLog[]
  movementAuthorized Movement[]          @relation("AuthorizedBy")
  custodySignatures  CustodyLog[]        @relation("OfficerSigned")
  medicationLogs     MedicationLog[]     @relation("DispenserBy")
  inventoryLogs      InventoryLog[]      @relation("LoggedBy")

  @@index([role])
  @@index([isActive])
  @@map("users")
}

// ─── PRISONERS ────────────────────────────────────────────────────────────────
model Prisoner {
  id          String    @id @default(cuid())
  prisonerId  String    @unique
  name        String
  age         Int
  crime       String
  sentence    String
  cellBlock   String
  cell        String
  status      String    @default("INCARCERATED") // INCARCERATED | RELEASED | TRANSFERRED | REMAND
  riskLevel   String    @default("MEDIUM")
  fingerprint String    @unique
  irisCode    String?
  entryDate   DateTime
  arrestDate  DateTime?
  releaseDate DateTime?
  photoUrl    String?
  notes       String?
  nationality String?
  nationalId  String?
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  incidents           Incident[]
  scans               ScanLog[]
  movements           Movement[]
  alerts              Alert[]
  visits              Visit[]
  medicalRecords      MedicalRecord[]
  vaccinations        Vaccination[]
  medications         MedicationLog[]
  rehabilitationLogs  RehabilitationLog[]
  warrants            Warrant[]
  sentences           SentenceRecord[]
  hearings            CourtHearing[]
  custodyLogs         CustodyLog[]
  biometricLogs       BiometricLog[]
  ecmsCases           EcmsCase[]

  @@index([cellBlock, cell])
  @@index([riskLevel])

}

// ─── NATIONAL INTEGRATIONS (NIRA / eCMS) ──────────────────────────────────────
// These records are cached locally after an online lookup so the system remains
// useful offline. Set cachedOffline = true once data is stored locally.

  id            String   @id @default(cuid())
  nationalId    String
  fullName      String?
  dob           String?

  matchedTo     String?  // prisoner prisonerId

  cacheExpiresAt DateTime? // set to 30 days after caching; prompt re-verification when online

  @@index([nationalId])
  @@index([cachedOffline])
  @@map("nira_lookups")
}

// ─── OFFLINE SYNC QUEUE ───────────────────────────────────────────────────────
// Stores all writes made while offline. The sync service drains this queue
// whenever connectivity is restored.
model SyncQueue {
  id           String    @id @default(cuid())
  deviceId     String
  entity       String
  operation    String    // CREATE | UPDATE | DELETE
  payload      String    // JSON blob of the full record
  localId      String?   // client-generated temp id
  serverIdMap  String?   // maps localId → real server id after sync
  status       String    @default("PENDING") // PENDING | SYNCED | CONFLICT | FAILED
  conflictData String?   // JSON: server record that conflicts with local change
  retryCount   Int       @default(0)     // incremented on each failed attempt
  lastError    String?   // last error message for debugging
  createdAt    DateTime  @default(now())
  syncedAt     DateTime?

  @@index([deviceId, status])
  @@index([status, createdAt])
  @@map("sync_queue")
}

// ─── DEVICE REGISTRY ──────────────────────────────────────────────────────────

model DeviceRegistry {
  id           String    @id @default(cuid())
  deviceId     String    @unique  // UUID generated on first app launch
  deviceName   String
  deviceType   String    @default("TABLET") // TABLET | DESKTOP | TERMINAL
  assignedTo   String?   // userId of primary user (optional)
  isAuthorized Boolean   @default(false)     // must be approved by ADMIN before syncing
  lastSeenAt   DateTime?
  lastSyncAt   DateTime?
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt

  @@index([isAuthorized])
  @@map("device_registry")
}

// ─── DATA PRIVACY (Uganda DPA 2019) ───────────────────────────────────────────
model DataRequest {
  id           String    @id @default(cuid())
  requestType  String    // ACCESS | RECTIFICATION | ERASURE | RESTRICTION | PORTABILITY | OBJECTION
  subjectName  String
  subjectId    String?   // prisonerId or nationalId
  requestedBy  String    // Name of person making request
  relationship String?   // Self | Legal Rep | Family
  status       String    @default("PENDING") // PENDING | IN_REVIEW | FULFILLED | REJECTED
  notes        String?
  resolution   String?
  dueDate      DateTime  // Must be resolved within 21 days per Uganda DPA
  resolvedAt   DateTime?
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt
  handledBy    String?   // userId of officer who resolved

  @@index([status])
  @@index([dueDate])
  @@map("data_requests")
}
